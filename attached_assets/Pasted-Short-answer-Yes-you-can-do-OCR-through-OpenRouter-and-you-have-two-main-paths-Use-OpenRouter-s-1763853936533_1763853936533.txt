import OpenAI from "openai";

const client = new OpenAI({
  apiKey: process.env.OPENROUTER_API_KEY!,
  baseURL: "https://openrouter.ai/api/v1",
});

export async function generateSurveyFromPdf(pdfUrl: string) {
  const completion = await client.chat.completions.create(
    {
      model: "moonshotai/kimi-k2:free", // or your surveyGeneration pool
      messages: [
        {
          role: "user",
          content: [
            {
              type: "text",
              text:
                "Read this training PDF and extract the key topics and learning " +
                "objectives, then draft a 15-question post-training evaluation survey.",
            },
            {
              type: "input_file",
              file_url: pdfUrl, // public URL or presigned URL of the PDF
            },
          ],
        },
      ],
      // üëá PDF plugin ‚Äì choose the extraction engine
      plugins: [
        {
          id: "file-parser",
          pdf: {
            engine: "pdf-text", // or "mistral-ocr" for scanned docs
          },
        },
      ],
    },
    {
      headers: {
        "HTTP-Referer": "https://evalia.app",
        "X-Title": "Evalia",
      },
    },
  );

  return completion.choices[0]?.message?.content ?? "";
}
